Q1: Enter a matrix of 4 rows and 2 columns from the user display it in matrix format?

        
#include <stdio.h>

int main() {
    int size1= 4; // 4 rows
    int size2 = 2; // 2 columns
    int i, j;

    int matrix[4][2];

    printf("Enter a matrix with 4 rows and 2 columns:\n");
    for (i = 0; i < size1; i++) {
        for (j = 0; j < size2; j++) {
            printf("Enter element on [%d,%d]: ", i, j);
            scanf("%d", &matrix[i][j]);
        }
    }

    printf("Matrix entered by the user:\n");
    for (i = 0; i < size1; i++) {
        for (j = 0; j < size2; j++) {
            printf("%4d", matrix[i][j]);
        }
        printf("\n");
    }

    return 0;
}
