Q5: Enter a Matrix from user, copy it in another matrix of same size?

        
#include <stdio.h>

int main() {
    int rows, cols, i, j;
    printf("Enter the number of Rows and Columns: ");
    scanf("%d %d", &rows, &cols);

    int mtrx[20][30], copied[20][30];

    printf("Enter the elements of the original matrix:\n");
    for (i = 0; i < rows; i++) {
        for (j = 0; j < cols; j++) {
            printf("Enter element on [%d,%d]: ", i, j);
            scanf("%d", &mtrx[i][j]);
            copied[i][j] = mtrx[i][j]; 
        }
    }

    printf("Original Matrix:\n");
    for (i = 0; i < rows; i++) {
        for (j = 0; j < cols; j++) {
            printf("%4d", mtrx[i][j]);
        }
        printf("\n");
    }

    printf("Copied Matrix:\n");
    for (i = 0; i < rows; i++) {
        for (j = 0; j < cols; j++) {
            printf("%4d", copied[i][j]);
        }
        printf("\n");
    }

    return 0;
}
