Q6: Enter the Matrix from user, multiply it with a scalar value and display the answer?

        
#include <stdio>
void main() {
    int rows, cols, i, j;
    int mtrx[20][30];
    int k;//scalar variable
    printf("Enter the number of Rows and Columns: ");
    scanf("%d %d", &rows, &cols);


    printf("Enter the elements of the matrix:\n");
    for (i = 0; i < rows; i++) {
        for (j = 0; j < cols; j++) {
            printf("Enter element on [%d,%d]: ", i, j);
            scanf("%d", &mtrx[i][j]);
        }
    }

    printf("Enter a scalar value to multiply with the matrix: ");
    scanf("%d", &k);

    printf("Original Matrix:\n");
    for (i = 0; i < rows; i++) {
        for (j = 0; j < cols; j++) {
            printf("%4d", mtrx[i][j]);
        }
        printf("\n");
    }

    // Multiply the matrix by the scalar
    for (i = 0; i < rows; i++) {
        for (j = 0; j < cols; j++) {
            mtrx[i][j] = mtrx[i][j] * k;
        }
    }

    printf("Matrix after multiplication with scalar:\n");
    for (i = 0; i < rows; i++) {
        for (j = 0; j < cols; j++) {
            printf(" %4d", mtrx[i][j]);
        }
        printf("\n");
    }

    return 0;
    
}
