Q11: Enter an array from user and find the duplicate values?

        
#include <stdio.h>
int main() {
int arr[20], i, j, size;
                    
printf("Enter how many elements: ");
scanf("%d", &size);
                
for (i = 0; i < size; i++) {
     printf("Enter %dth element: ", i);
    scanf("%d", &arr[i]);
                    }
                
printf("Duplicate values in the array are: ");
                    
for (i = 0; i < size; i++) {
    for (j = i + 1; j < size; j++) {
        if (arr[i] == arr[j]) {
            // Found a duplicate value
            printf("%d ", arr[i]);
            break; // Break to avoid printing the same duplicate multiple times
        }
    }
}
}

Output:-


Enter how many elements: 5
Enter 0th element: 10
Enter 1th element: 20
Enter 2th element: 10
Enter 3th element: 20
Enter 4th element: 30
Duplicate values in the array are: 10 20
