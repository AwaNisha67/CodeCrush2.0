#include <stdio.h>

int main() {
 int arr[20], i, j, size;
            
 printf("Enter how many elements: ");
 scanf("%d", &size);
            
 for (i = 0; i < size; i++) {
     printf("Enter %dth element: ", i);
     scanf("%d", &arr[i]);
 }
            
 // Remove duplicate values by replacing them with zero
 for (i = 0; i < size; i++) {
     for (j = i + 1; j < size; j++) {
         if (arr[i] == arr[j]) {
             arr[j] = 0;
         }
     }
 }
            
printf("Array after removing duplicates and filling with zeros:\n");
            
for (i = 0; i < size; i++) {
     printf("%d ", arr[i]);
}
            
return 0;
}

Output:-

Enter 0th element: 10
Enter 1th element: 20

Enter 4th element: 10
