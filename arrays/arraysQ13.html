Enter an array from user and count the duplicate values?

        
#include <stdio.h>

int main() {
    int arr[20], i, j, size, count = 0;
                
    printf("Enter how many elements: ");
    scanf("%d", &size);
                
    for (i = 0; i <size; i++) {
        printf("Enter %dth element: ", i);
         scanf("%d", &arr[i]);
    }
                
    for (i = 0; i < size; i++) {
        for (j = i + 1; j < size; j++) {
            if (arr[i] == arr[j]) {
                                
                count++;
                break;
            }
        }
    }
                
    printf("Number of duplicate values in the array: %d\n", count);
                
    return 0;
}
