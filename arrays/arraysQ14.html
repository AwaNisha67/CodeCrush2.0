Q14: Enter an array from user and print all possible combinations of the elements? 

        
#include<stdio.h>
    int main(){

        printf("Enter size of the array: ");

        printf("Enter values of your array: ");
        for(i=0; i<size; i++){
            printf("\n%d element",i);
            scanf("%d", &arr[i]);
        }
        printf("Your values are: ");
        for(i=0; i<size; i++){
            printf(" %d ", arr[i]);
        }
        printf("\n");

            for(j=0; j<size; j++){
                    printf(" %d %d\n ", arr[i], arr[j]);

         }

    }
