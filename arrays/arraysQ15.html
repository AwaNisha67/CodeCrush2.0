Q15: Enter an array from user and print all the palindrone numbers

        
#include <stdio.h>
int pallindrome(int arr[], int size) {
    int i, j, count = 0;
    for(i = 0; i<size; i++) {
        for(j=size-1;j>=0;j--)
        if(arr[i] != arr[j]) {
            count = 1;
            break;
         }

     }
    if (count == 1)
         return 0;
     else
         return 1;
}

    int arr[69],size,i,result;
