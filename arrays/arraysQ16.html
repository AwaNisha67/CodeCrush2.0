Enter two arrays from the user and check they are same or not?


#include <stdio.h>
int main() {
int arr1[20], arr2[20], i, size1,size2;
        
printf("Enter how many (1st)elements in array: ");
scanf("%d", &size1);
        
printf("Enter elements for the first array:\n");
for (i = 0; i < size1; i++) {
    printf("Enter %dth element: ", i);
    scanf("%d", &arr1[i]);
}
printf("Enter how (2nd)many elements in array: ");
scanf("%d", &size2);
      
printf("Enter elements for the second array:\n");
for (i = 0; i < size2; i++) {
    printf("Enter %dth element: ", i);
    scanf("%d", &arr2[i]);
}
if(size1 != size2){
    printf("\nArrays are not of same size");

}  
else{
int areEqual = 1; 
// assume kar rhe ki array already equal hi
        
for (i = 0; i < size1; i++) {
    if (arr1[i] != arr2[i]) {
        areEqual = 0;  // Arrays are not equal
        break;
    }
}
        
if (areEqual == 1) {
    printf("The two arrays are the same.\n");
} else {
    printf("The two arrays are not the same.\n");
}
}        
 return 0;
    
}
