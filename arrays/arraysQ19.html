Q19: Find the local maxima in the array?


#include 

void takeInput(int arr[])
{
    for (int i = 0; i < 10; i++)
    {
        printf("Enter the value for the %d position: ", i + 1);
        scanf("%d", &arr[i]);
    }
}

int FindLocalMaxima(int arr[], int length)
{
    // first and last
    if (arr[0] > arr[1])
    {
        printf("%d at index 0 is a local maximum\n", arr[0]);
    }
    if (arr[length - 1] > arr[length - 2])
    {
        printf("%d at index %d is a local maximum\n", arr[length - 1], length - 1);
    }
    for (int i = 1; i < length - 1; i++)
    {
        if ((arr[i] > arr[i - 1]) && (arr[i] > arr[i + 1]))
        {
            printf("%d at index %d is a local maximum\n", arr[i], i);
        }
    }
}

int main()
{
    int arr[10];
    takeInput(arr);
    FindLocalMaxima(arr, 10);
    return 0;
}
