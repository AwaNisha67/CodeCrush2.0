//  Enter an array from user and copy it into another array in reverse order?
#include<stdio.h>
void main(){
    int arr[]={10,20,30,60,80};
    int brr[6],size1=5,i,j;
   for(i=size1-1;i>=0;i--){
       brr[i]=arr[(size1-1)-i];
   }
    printf("Copied array: ");
    for(i=0;i<size1;i++){
        printf("%d ",brr[i]);
    }
    printf("\n");
}
