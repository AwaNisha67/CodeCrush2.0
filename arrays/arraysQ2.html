Enter an array from user and print the array in reverse order ?

        
// Enter an array from user and print the array in reverse order ?
#include<stdio.h>
void main(){
    int arr[69],size,i;
    printf("\nEnter the size of the array: ");
    scanf("%d",&size);
    
    printf("\nEnter elements of the array: ");
    for(i=0;i<size;i++){
        printf("\n%dth element: ",i);
        scanf("%d",&arr[i]);
    }
    printf("\nArray in reversed-order:\n");
    for(i=size-1;i>=0;i--){
        printf("%4d",arr[i]);
    }
}

OutPut:-

        
Enter the size of the array: 5
Enter elements of the array: 
0th element: 10
1th element: 20
2th element: 3

4th element: 50
Array in reversed-order:
  50  40   3  20  10
