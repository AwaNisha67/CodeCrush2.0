Enter an array from user and copy the array into second array ?

        
#include<stdio.h>
void main(){
    int arr[69],brr[69],size,i;
    printf("\nEnter the size of the array: ");
    scanf("%d",&size);
    
    printf("\nEnter elements of the array: ");
    for(i=0;i<size;i++){
        printf("\n%dth element: ",i);
        scanf("%d",&arr[i]);
    }
    printf("Array entered:\n ");
    for(i=0;i <size;i++){
        printf("%4d",arr[i]);
    }
    for(i=0;i<size;i++){
        brr[i]=arr[i];
    }
     printf("\nCopied Array:\n ");
    for(i=0;i<size;i++){
        printf("%4d",brr[i]);
    }
}
