Enter an array from user and find maximum and minimum value?

        
#include<stdio.h>
void main(){
    int arr[10],size,i,max,min;
    printf("Enter the size of the arrays (1-10): ");
    scanf("%d",&size);
    
    printf("Enter elements in the array:\n ");
    for(i=0;i<size;i++){
        printf("%d element",i+1);
        scanf("%d",&arr[i]);
    }
    printf("Maximum value in array: ");
    max=arr[0];
    for(i=0;i<size;i++){
        if(arr[i]>max){
            max=arr[i];
        }
    }
    printf("%d ",max);
    
    printf("\nMinimum value in array: ");
    min=arr[0];
    for(i=0;i<size;i++){
        if(arr[i]<min){
            min=arr[i];
        }
    }
    printf("%d ",min);
}
