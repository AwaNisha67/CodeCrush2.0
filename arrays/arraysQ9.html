Enter an array from user and find the k-th smallest value?

        
#include <stdio.h>

int main() {
int arr[20], i, size, k,temp;
                
printf("Enter how many elements: ");
scanf("%d", &size);
                
for (i = 0; i < size; i++) {
     printf("Enter %dth element: ", i);
     scanf("%d", &arr[i]);
}
                
printf("Enter the value(index) of k to find the k-th smallest element: ");
scanf("%d", &k);
                
if (k > 0 && k <= size) {
    for (i = 0; i < size-1; i++) {
        for (int j = 0; j<size-i-1; j++) {
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
             }
         }
     }
printf("The %d-th smallest element is: %d\n", k, arr[k - 1]);
else {
    printf("Invalid value\n");
    }
}
