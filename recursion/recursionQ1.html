1. CALCULATE FACTORIAL OF A NUMBER USING RECURSION


#include<stdio.h>
int fact(int n)
{
    if(n==0)
    {
        return 1;
    }
    else
    {
        return n*fact(n-1);
    }
}
void main()
{
    unsigned int n;
    printf("enter a positive number ");
    scanf("%u",&n);
    printf("factorial is %d ",fact(n));
}

Output:-


enter a positive number 5
factorial is 120
