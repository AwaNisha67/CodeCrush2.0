CALCULATE THE SUM OF NATURAL NUMBERS TILL 10 USING RECURSION.


        #include<stdio.h>
        int main()
        {
            int n=10;
            printf("Sum is : %d",sum(n));
            return 0;
            }
            //recursive function to calculate the sum of natural numbers
            int sum(int num)
            {
                if (num==0)
                return 0;
                else
                return (num + sum(num-1));
            }

OutPut:-


        Sum is : 55
