CONVERT A NUMBER INTO BINARY USING RECURSION


#include<stdio.h>
 void main()
 {
     int n;
     printf("Enter a decimal number: ");
     scanf("%d",&n);
     convert(n);
     }
     void convert(int n)
     {
         if (n > 0)
         {
             convert(n / 2); //recursive call to divide the number by two
             printf("%d ", n % 2); //prints remainder of division
             }
             else
             return ;
}
