Q7. ENTER A INTEGER ARRAY FROM THE USER, ENTER AN INTEGER VALUE AND PERFORM BINARY SEARCH ALGORITHM THROUGH RECURSION

        
        
    #include <stdio.h>  
int recursiveBinarySearch(int arr[], int start, int end, int key)    
{    
        int mid;    
        if(start <= end )     
        {   mid = (start + end)/2;    
            if(arr[mid] == key)    
            {                 
                return mid+1;    
        }    
        else if(arr[mid] < key)     
        {  
                return recursiveBinarySearch(arr, mid+1, end, key);    
        }    
        else     
        {  
                return recursiveBinarySearch(arr, start, mid-1, key);    
        }              }    
    return -1;     }   
int main() {  
      int arr[50] ;  
      int key,size,i,temp;

    scanf("%d",&arr[i]);}
  printf("Enter element to be search");
  scanf("%d",&key);
  
  temp=recursiveBinarySearch(arr,0,size-1,key);
  if (temp == -1)  
  printf("\nElement is not present in the array");
