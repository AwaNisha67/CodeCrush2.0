Q2. ENTER A INTEGER ARRAY FROM THE USER AND PERFORM SELECTION SORT ALGORITHM THROUGH RECURSION. ALSO CALCULATE THE WORST AND BEST COMPLEXITY WITH EXACT TIME.

        
            #include <stdio.h>

                void selectionSort(int arr[], int size, int current){
                    int minIndex = current;
                
                    for (int j = current + 1; j < size; j++) {
                        if (arr[j] < arr[minIndex]) {
                            minIndex = j;
                        }
                    }
                    int temp = arr[current];
                    arr[current] = arr[minIndex];
                    arr[minIndex] = temp;
                
                    selectionSort(arr, size, current + 1);
                }
                
                int main() {
                    int arr[5] = {20, 4, 50, 6, 9};
                    selectionSort(arr, 5, 0);
                    
                    printf("Sorted array: ");
                    for (int i = 0; i < 5; i++) {
                        printf("%d ", arr[i]);
                    }
                    printf("\n");
                
                }
