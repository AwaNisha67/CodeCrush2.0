Q4. ENTER A INTEGER ARRAY FROM THE USER AND PERFORM BUBBLE SORT ALGORITHM THROUGH RECURSION. ALSO CALCULATE THE WORST AND BEST COMPLEXITY WITH EXACT TIME.

        
            #include <stdio.h>

                void bubbleSort(int arr[], int size) {
                   if (size <= 1) {
                        return;  // if size is less than one the element is already sorted
                    } 
                    for (int i = 0; i < size - 1; i++) {
                        if (arr[i] > arr[i + 1]) {
                            // Swap the elements
                            int temp = arr[i];
                            arr[i] = arr[i + 1];
                            arr[i + 1] = temp;
                        }
                    }
                    bubbleSort(arr, size - 1);
                }
                
                int main() {
                    int size;
                    int arr[50];
                
                    printf("Enter the size of the array: ");
                    scanf("%d", &size);
                
                    printf("Enter the elements:\n");
                    for (int i = 0; i < size; i++) {
                        printf("Element %d: ", i + 1);
                        scanf("%d", &arr[i]);
                    }
                
                    bubbleSort(arr, size);
                
                    printf("Sorted elements are: ");
                    for (int i = 0; i < size; i++) {
                        printf("%d ", arr[i]);
                    }
                }
