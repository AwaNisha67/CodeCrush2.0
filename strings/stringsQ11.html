Q11: ENTER THE STRING FROM USER AND CHECK ITS PALINDROME OR NOT?

        
            #include <stdio.h>

int main() {
    char str1[69];
    int len = 0;
    int isPalindrome = 1; // Assumimg ki it's a palindrome by default   
    printf("Enter a string: ");
    scanf("%s", str1);

    // Calculate the length of the entered string
    while (str1[len] != '\0') {
        len++;
    }
    for (int i = 0; i < len / 2; i++) {
        if (str1[i] != str1[len - 1 - i]) {
            isPalindrome = 0;
            break;
        }
    }
                    
if (isPalindrome == 1) {
    printf("The entered string is a palindrome.\n");
} else {
    printf("The entered string is not a palindrome.\n");
}

 return 0;
}
