Q18: ENTER THE STRING FROM THE USER AND CONVERT INTO AN INTEGER VALUE?

        
#include <stdio.h>

    char str1[69];
    int len = 0;
    printf("\nEnter a string: ");
    scanf("%s", str1);

    // Calculate the length of the entered string
    while (str1[len] != '\0') {
        len++;
    }
   printf("Length of the entered string: %d\n",len);

    for (int i = 0; str1[i]; i++) {

    }
}
