Q2: ENTER THE STRING FROM USER AND CONVERT IN LOWERCASE ?

        
#include<stdio.h>
void main(){
    char s1[20];
    int i;
    printf("Enter a string:(in UPPERCASE):");
    scanf("%s",s1);
    
    for(i=0;s1[i];i++){
        if(s1[i]>=65 && s1[i]<=90){
            printf("%c",s1[i]+32);
        }
         else{
            printf("%c",s1[i]);
        }
    }
    
}
