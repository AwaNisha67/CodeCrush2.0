Q22: ENTER THE STRING FROM THE USER, SORT THE STRING IN ASCENDING ORDER AND PRINT IT.

        
#include <stdio.h>

int main() {
    char str1[69];
    int len = 0;

    printf("Enter a string: ");
    scanf("%s", str1);

    // Calculate the length of the entered string
    while (str1[len] != '\0') {
        len++;
    }

    // Bubble sort for sorting string
    for (int i = 0; i < len - 1; i++) {
        for (int j = 0; j < len - i - 1; j++) {
            if (str1[j] > str1[j + 1]) {
              
                char temp = str1[j];
                str1[j] = str1[j + 1];
                str1[j + 1] = temp;
            }
        }
    }

    printf("Sorted String in Ascending Order: %s\n", str1);

}

Output:-


Enter a string: yelllow
Sorted String in Ascending Order: elllowy
