Q9: ENTER THE STRING FROM USER AND COPY IT INTO ANOTHER STRING VARIABLE?

        
#include <stdio.h>

int main() {
    char str1[100]; 
    char str2[100];   
    char ch;
    int len=0,i;

    printf("\nEnter a string: ");
    scanf("%s",str1);
    
    for(int i=0;str1[i];i++){
        len++;
    }
    printf("\nLength of the entered String is: %d",len);
    for (i = 0; i <= len; i++) {
        str2[i] = str1[i];
    }

    printf("\nOriginal String: %s\n", str1);
    printf("Copied String: %s\n", str2);

    return 0;
}
